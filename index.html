import turtle
import math

def setup_turtle():
    """Initialize turtle with standard settings"""
    t = turtle.Turtle()
    t.speed(0) # Fastest speed
    screen = turtle.Screen()
    screen.title("Turtle Graphics Assignment")
    return t, screen

def draw_rectangle(t, width, height, fill_color=None):
    """Draw a rectangle with optional fill"""
    if fill_color:
        t.fillcolor(fill_color)
        t.begin_fill()
    for _ in range(2):
        t.forward(width)
        t.left(90)
        t.forward(height)
        t.left(90)
    if fill_color:
        t.end_fill()

def draw_square(t, size, fill_color=None):
    """Draw a square with optional fill"""
    draw_rectangle(t, size, size, fill_color)

def draw_triangle(t, width, height, fill_color=None):
    """Draw an isosceles triangle for the roof"""
    if fill_color:
        t.fillcolor(fill_color)
        t.begin_fill()
    t.forward(width)
    t.left(135)
    t.forward(math.sqrt(2)*(width/2))
    t.left(90)
    t.forward(math.sqrt(2)*(width/2))
    t.left(135)
    if fill_color:
        t.end_fill()

def draw_circle(t, radius, fill_color=None):
    """Draw a circle with optional fill"""
    if fill_color:
        t.fillcolor(fill_color)
        t.begin_fill()
    t.circle(radius)
    if fill_color:
        t.end_fill()

def jump_to(t, x, y):
    """Move turtle without drawing"""
    t.penup()
    t.goto(x, y)
    t.pendown()

def draw_tree(t, x, y, trunk_width=20, trunk_height=60, foliage_radius=35, trunk_color="sienna", foliage_color="forestgreen"):
    """Draw a simple tree at (x, y) with the foliage circle touching the trunk top."""
    # Draw trunk
    jump_to(t, x, y)
    t.setheading(0)
    draw_rectangle(t, trunk_width, trunk_height, fill_color=trunk_color)
    # Draw foliage so it touches the trunk
    foliage_x = x + trunk_width / 2
    foliage_y = y + trunk_height
    jump_to(t, foliage_x, foliage_y - foliage_radius) # Move to bottom of foliage circle
    t.setheading(0)
    draw_circle(t, foliage_radius, fill_color=foliage_color)

def draw_scene(t):
    # Set background color
    screen = t.getscreen()
    screen.bgcolor("skyblue")

    # --- Draw House 1 (Red Rectangle) ---
    house_width = 200
    house_height = 120
    house_x = -100
    house_y = -100
    jump_to(t, house_x, house_y)
    t.setheading(0)
    draw_rectangle(t, house_width, house_height, fill_color="red")

    # --- Draw Roof 1 (Brown Triangle) ---
    jump_to(t, house_x, house_y + house_height)
    t.setheading(0)
    draw_triangle(t, house_width, 60, fill_color="saddlebrown")

    # --- Draw Door 1 (Yellow Rectangle) ---
    door_width = 40
    door_height = 60
    door_x = house_x + house_width/2 - door_width/2
    door_y = house_y
    jump_to(t, door_x, door_y)
    t.setheading(0)
    draw_rectangle(t, door_width, door_height, fill_color="yellow")

    # --- Draw Windows 1 (Blue Squares) ---
    window_size = 35
    # Left window
    win1_x = house_x + 30
    win1_y = house_y + house_height/2
    jump_to(t, win1_x, win1_y)
    t.setheading(0)
    draw_square(t, window_size, fill_color="deepskyblue")
    # Right window
    win2_x = house_x + house_width - window_size - 30
    win2_y = win1_y
    jump_to(t, win2_x, win2_y)
    t.setheading(0)
    draw_square(t, window_size, fill_color="deepskyblue")

    # --- Draw House 2 (Orange) ---
    house2_width = 150
    house2_height = 90
    house2_x = 120
    house2_y = -90
    jump_to(t, house2_x, house2_y)
    t.setheading(0)
    draw_rectangle(t, house2_width, house2_height, fill_color="orange")

    # --- Draw Roof 2 (Dark Red Triangle) ---
    jump_to(t, house2_x, house2_y + house2_height)
    t.setheading(0)
    draw_triangle(t, house2_width, 45, fill_color="maroon")

    # --- Draw Door 2 (Brown Rectangle) ---
    door2_width = 32
    door2_height = 45
    door2_x = house2_x + house2_width/2 - door2_width/2
    door2_y = house2_y
    jump_to(t, door2_x, door2_y)
    t.setheading(0)
    draw_rectangle(t, door2_width, door2_height, fill_color="brown")

    # --- Draw Windows 2 (Light Blue Squares) ---
    window2_size = 28
    # Left window
    win21_x = house2_x + 20
    win21_y = house2_y + house2_height/2
    jump_to(t, win21_x, win21_y)
    t.setheading(0)
    draw_square(t, window2_size, fill_color="lightblue")
    # Right window
    win22_x = house2_x + house2_width - window2_size - 20
    win22_y = win21_y
    jump_to(t, win22_x, win22_y)
    t.setheading(0)
    draw_square(t, window2_size, fill_color="lightblue")

    # --- Draw Sun (Yellow Circle) ---
    sun_radius = 40
    sun_x = 180
    sun_y = 130
    jump_to(t, sun_x, sun_y)
    t.setheading(0)
    draw_circle(t, sun_radius, fill_color="yellow")

    # --- Draw Clouds (White Ovals) ---
    # Cloud 1
    jump_to(t, -220, 170)
    t.setheading(0)
    t.fillcolor("white")
    t.begin_fill()
    for _ in range(2):
        t.circle(30,90)
        t.circle(30//2,90)
    t.end_fill()
    # Cloud 2
    jump_to(t, -160, 140)
    t.setheading(0)
    t.fillcolor("white")
    t.begin_fill()
    for _ in range(2):
        t.circle(22,90)
        t.circle(22//2,90)
    t.end_fill()

    # --- Draw Tree 1 (Left side, away from houses and clouds) ---
    draw_tree(t, x=-300, y=-120, trunk_width=20, trunk_height=60, foliage_radius=32, trunk_color="sienna", foliage_color="forestgreen")

    # --- Draw Tree 2 (Far right, away from houses and sun) ---
    draw_tree(t, x=320, y=-120, trunk_width=20, trunk_height=60, foliage_radius=32, trunk_color="sienna", foliage_color="forestgreen")

def main():
    t, screen = setup_turtle()
    draw_scene(t)
    screen.mainloop()

if __name__ == "__main__":
    main()

# Text Analysis with SpaCy

import spacy

# Load the English NLP model
nlp = spacy.load("en_core_web_sm")

# Example text
text = "SpaCy is an open-source library for advanced Natural Language Processing in Python."

# Process the text
doc = nlp(text)

# Tokenization
print("Tokens:")
for token in doc:
    print(token.text)

# Part-of-speech tagging
print("\nPart-of-speech tags:")
for token in doc:
    print(f"{token.text}: {token.pos_}")

# Named Entity Recognition
print("\nNamed Entities:")
for ent in doc.ents:
    print(f"{ent.text} ({ent.label_})")
